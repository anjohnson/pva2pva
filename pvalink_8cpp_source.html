169 {
170  bool doscan = false;
171  // check if we actually need to scan anything
172  FOREACH(links_t::const_iterator, it, end, links) {
173  pvaLink* L = *it;

177  {
178  doscan = true;

193 void pvaLinkChannel::scan(void* arg, epicsJobMode mode)
194 {
195  pvaLinkChannel *selfraw = (pvaLinkChannel*)arg;
196  if(mode!=epicsJobModeRun) return; // we will cleanup later
197  pvaGlobal_t::Scan myscan;
198  try {
199  if(pvaLinkDebug>3) std::cerr<<"pvaLink scan "<<selfraw->name<<"\n";
200 
201  std::tr1::shared_ptr<pvaLinkChannel> self;
202 
203  Guard G(selfraw->lock);
204 
205  selfraw->scanself.swap(self); // we take over ref, to keep channel alive, and allow re-queue
206  assert(self.get()==selfraw); // if scanself wasn't set, then the channel may be free'd
207 
208  myscan.chan = self; // store a weak ref
209 
210  links_t links(self->links); // TODO: avoid copy if set not changing
211 
212  bool usecached = self->scanatomic && !!self->chanmon;
213  myscan.usecached = usecached;
214  if(usecached) {
215  if(pvaLinkDebug>4) std::cerr<<"populate cache\n";
216  FOREACH(links_t::const_iterator, it, end, links) {
217  pvaLink *link = *it;
218  link->get(link->atomcache);
219  if(pvaLinkDebug>4)
220  std::cerr<<"== "<<self->name<<"."<<link->field<<" "<<link->valueS<<"\n";
221  }
222  }
223 
224  pvaGlobal->scanmagic.set(usecached ? &myscan : NULL);
225 
226  {
227  UnGuard U(G);
228  // we may scan a record after the originating link is re-targeted
229 
230  FOREACH(links_t::const_iterator, it, end, links) {
231  pvaLink *link = *it;
232  dbCommon *prec=link->plink->precord;
233 
234  if ((link->linkmods & pvlOptCP) ||
235  ((link->linkmods & pvlOptCPP) && prec->scan == 0))
236  {
237  DBScanLocker L(prec);
238  if(pvaLinkDebug>3) std::cerr<<prec->name<<" PVA link scan\n";
239  dbProcess(prec);
240  }
241  }
242  }
243  // another scan may be queued by this point
244 
245  if(usecached) {
246  FOREACH(links_t::const_iterator, it, end, links) {
247  pvaLink *link = *it;
248  link->atomcache.clear();
249  }
250  }
251 
252  }catch(std::exception& e){
253  errlogPrintf("%s: pvaLink exception while processing: %s\n", selfraw->name.c_str(), e.what());
254  // what to do?
255  }
256  pvaGlobal->scanmagic.set(NULL);
257 }

355 long pvaGetValue(DBLINK *plink, short dbrType, void *pbuffer,

358  TRY {

360  const void *buf;

362  if(self->atomcache.scalar) {

364  count = std::min((size_t)1u, count);
365  } else {
366  buf = self->atomcache.valueA.data();
367  count = std::min(self->atomcache.valueA.size(), count);
368  }
369 
370  pvd::castUnsafeV(count, DBR2PVD(dbrType), pbuffer, self->atomcache.etype, buf);
371 // if(dbrType==DBF_DOUBLE) {
372 // std::cerr<<"get from cache "<<*(double*)pbuffer<<"\n";
373 // }

375  return 0;

377 
378  Guard G(self->lchan->lock);
379  if(self->valueA) {

381  self->valueA->getAs<const void>(arrval);
382 
383  long nelem = std::min(*pnRequest, (long)arrval.size());
384 
385  pvd::castUnsafeV(nelem, DBR2PVD(dbrType), pbuffer, arrval.original_type(), arrval.data());
386  if(pnRequest) *pnRequest = nelem;
387 
388  } else if(self->valueS) {
389 
390  switch(dbrType) {
391 #define CASE(BASETYPE, PVATYPE, DBFTYPE, PVACODE) case DBR_##DBFTYPE: *((epics##BASETYPE*)pbuffer) = self->valueS->getAs<epics##BASETYPE>(); break;

403  break;
404  default:
405  throw std::runtime_error("putValue unsupported DBR code");
406  }
407  if(pnRequest) *pnRequest = 1;
408 
409  if(dbrType==DBF_DOUBLE)
410  std::cerr<<"get direct "<<*(double*)pbuffer<<"\n";
411  } else {
412  return -1;
413  }
414  return 0;
415  }CATCH(pvaIsConnected)
